### 🔴 P0 - 必须实现（核心功能缺失）

#### 1. FPS 实时监控
**当前状态**: 返回硬编码 60  
**用户价值**: ⭐⭐⭐⭐⭐ 用户需要知道壁纸是否流畅  
**技术可行性**: ✅ 使用 `CADisplayLink` 标准实现  
**实现成本**: 低（1-2 小时）  
**竞品对比**: 所有动态壁纸软件的标配功能  

**实现方案**:
```swift
class PerformanceMonitor {
    private var displayLink: CADisplayLink?
    private var frameCount = 0
    private var lastTimestamp: CFTimeInterval = 0
    private(set) var currentFPS = 0
    
    func startMonitoring() {
        displayLink = CADisplayLink(target: self, selector: #selector(tick))
        displayLink?.add(to: .main, forMode: .common)
    }
    
    @objc private func tick(link: CADisplayLink) {
        frameCount += 1
        if link.timestamp - lastTimestamp >= 1.0 {
            currentFPS = frameCount
            frameCount = 0
            lastTimestamp = link.timestamp
        }
    }
}
```

**建议**: 立即实现，这是性能监控的基础

#### 3. 全屏应用时暂停
**当前状态**: 检测逻辑错误（只检查自己的窗口）  
**用户价值**: ⭐⭐⭐⭐⭐ 用户全屏看视频/玩游戏时必须暂停壁纸  
**技术可行性**: ✅ 使用 `CGWindowListCopyWindowInfo`  
**实现成本**: 中（4-6 小时，需要处理多显示器）  
**竞品对比**: 所有动态壁纸软件的核心功能  

**实现方案**:
```swift
private func hasFullscreenApp() -> Bool {
    guard let windowList = CGWindowListCopyWindowInfo([.optionOnScreenOnly, .excludeDesktopElements], kCGNullWindowID) as? [[String: Any]] else {
        return false
    }
    
    for window in windowList {
        // 检查窗口是否全屏（bounds == screen bounds）
        if let bounds = window[kCGWindowBounds as String] as? [String: CGFloat],
           let layer = window[kCGWindowLayer as String] as? Int,
           layer == 0 {  // 普通窗口层级
            // 比较窗口大小和屏幕大小
            for screen in NSScreen.screens {
                let screenFrame = screen.frame
                if abs(bounds["Width"]! - screenFrame.width) < 10 &&
                   abs(bounds["Height"]! - screenFrame.height) < 10 {
                    return true
                }
            }
        }
    }
    return false
}
```

**注意事项**:
- 需要处理多显示器（某个屏幕全屏时，其他屏幕的壁纸是否暂停？）
- 需要排除 PlumWallPaper 自己的窗口
- 需要处理 macOS 原生全屏（绿色按钮）和应用自定义全屏

**建议**: 高优先级，修复现有错误实现

```swift
struct AppRule: Codable {
    let bundleIdentifier: String
    let appName: String
    let action: RuleAction
}

enum RuleAction: String, Codable {
    case pause = "pause"
    case mute = "mute"
    case limitFPS30 = "fps30"
    case limitFPS15 = "fps15"
    case none = "none"
}
```

**实现方案**:
```swift
class AppRuleManager {
    private var rules: [AppRule] = []
    
    func checkRunningApps() -> RuleAction? {
        let running = NSWorkspace.shared.runningApplications
        for app in running {
            if let bundleId = app.bundleIdentifier,
               let rule = rules.first(where: { $0.bundleIdentifier == bundleId }) {
                return rule.action
            }
        }
        return nil
    }
}
```

**建议**: 高优先级，这是"运行高负载应用时暂停"的完整版本

**当前状态**: 返回硬编码 0.0  
**用户价值**: ⭐⭐⭐ 用户想知道壁纸是否占用太多 GPU  
**技术可行性**: ⚠️ macOS 无直接 API，只能估算  
**实现成本**: 中（4-6 小时）  

**实现方案**（估算法）:
```swift
// 通过帧时间估算 GPU 负载
class PerformanceMonitor {
    private var frameTime: CFTimeInterval = 0
    
    func getGPULoad() -> Double {
        // 假设 60fps 时每帧 16.67ms
        let targetFrameTime = 1.0 / 60.0
        let load = min(frameTime / targetFrameTime * 100, 100)
        return load
    }
    
    @objc private func tick(link: CADisplayLink) {
        frameTime = link.targetTimestamp - link.timestamp
        // ...
    }
}
```

**注意**: 这不是真实的 GPU 使用率，只是"渲染负载"估算

**UI 改名**:
